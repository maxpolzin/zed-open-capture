188
189 sum += offset;
190 double avg_offset=sum/(++count);

192 std::stringstream avgOffsetStr;
193 avgOffsetStr << std::fixed << std::setprecision(9) << std::showpos << "Avg timestamp offset: " << avg_offset << " sec";

198 cv::putText( frameData, "Inertial sensor data:", cv::Point(display_resolution.width/2,20),cv::FONT_HERSHEY_SIMPLEX, 0.6, cv::Scalar(241, 240,236));

200 cv::putText( frameData, imuGyroStr, cv::Point(display_resolution.width/2+15, 62),cv::FONT_HERSHEY_SIMPLEX, 0.6, cv::Scalar(241, 240,236));
201 imuMutex.unlock();
202
203 // Resize Image for display
204 cv::resize(frameBGR, frameBGRDisplay, display_resolution);
205 // Display image
206 cv::imshow( "Stream RGB", frameDisplay );
207 }

229// Sensor acquisition runs at 400Hz, so it must be executed in a different thread
230void getSensorThreadFunc(sl_oc::sensors::SensorCapture* sensCap)
231{
232 // Flag to stop the thread
233 sensThreadStop = false;
234
235 // Previous IMU timestamp to calculate frequency
236 uint64_t last_imu_ts = 0;
237
238 // Infinite data grabbing loop
239 while(!sensThreadStop)
240 {
241 // ----> Get IMU data
242 const sl_oc::sensors::data::Imu imuData = sensCap->getLastIMUData(2000);
243
244 // Process data only if valid
245 if(imuData.valid == sl_oc::sensors::data::Imu::NEW_VAL ) // Uncomment to use only data syncronized with the video frames
246 {
247 // ----> Data info to be displayed
248 std::stringstream timestamp;
249 std::stringstream accel;
250 std::stringstream gyro;
251
252 timestamp << std::fixed << std::setprecision(9) << "IMU timestamp: " << static_cast<double>(imuData.timestamp)/1e9<< " sec" ;
253 if(last_imu_ts!=0)
254 timestamp << std::fixed << std::setprecision(1) << " [" << 1e9/static_cast<float>(imuData.timestamp-last_imu_ts) << " Hz]";
255 last_imu_ts = imuData.timestamp;
256
257 accel << std::fixed << std::showpos << std::setprecision(4) << " * Accel: " << imuData.aX << " " << imuData.aY << " " << imuData.aZ << " [m/s^2]";
258 gyro << std::fixed << std::showpos << std::setprecision(4) << " * Gyro: " << imuData.gX << " " << imuData.gY << " " << imuData.gZ << " [deg/s]";
259 // <---- Data info to be displayed
260
261 // Mutex to not overwrite data while diplaying them
262 imuMutex.lock();
263
264 imuTsStr = timestamp.str();
265 imuAccelStr = accel.str();
266 imuGyroStr = gyro.str();
267
268 // ----> Timestamp of the synchronized data
269 if(imuData.sync)
270 {
271 mcu_sync_ts = imuData.timestamp;
272 }
273 // <---- Timestamp of the synchronized data
274
275 imuMutex.unlock();
276 }
277 // <---- Get IMU data
278 }
279}
