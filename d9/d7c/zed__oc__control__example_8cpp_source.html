110// The main function
111int main(int argc, char *argv[])
112{
113 sl_oc::VERBOSITY verbose = sl_oc::VERBOSITY::INFO;
114
115 // ----> Set Video parameters
116 sl_oc::video::VideoParams params;
117 params.res = sl_oc::video::RESOLUTION::HD2K;
118 params.fps = sl_oc::video::FPS::FPS_15;
119 params.verbose = verbose;
120 // <---- Set Video parameters
121
122 // ----> Create Video Capture
123 sl_oc::video::VideoCapture cap(params);
124 if( !cap.initializeVideo(-1) )
125 {
126 std::cerr << "Cannot open camera video capture" << std::endl;
127 std::cerr << "See verbosity level for more details." << std::endl;
128
129 return EXIT_FAILURE;
130 }
131 std::cout << "Connected to camera sn: " << cap.getSerialNumber() << std::endl;
132 // <---- Create Video Capture
133
134 // ----> Create rendering window
135 cv::namedWindow(win_name);
136 cv::setMouseCallback(win_name, handleMouse);
137 // <---- Create rendering window
138
139 // Reset all the controls to default values
140 resetControls(cap);
141
142 // Set the default camera control setting
143 setActiveControl(cap, Brightness );
144
145 // Update the values for all the controls
146 updateAllCtrlValues(cap);
147
148 uint64_t last_ts=0;
149 uint16_t not_a_new_frame = 0;
150 int frame_timeout_msec = 100;
151
152 // Infinite video grabbing loop
153 while (1)
154 {
155 // 3) Get last available frame
156 const sl_oc::video::Frame frame = cap.getLastFrame(frame_timeout_msec);
157 img_w = frame.width;
158 img_h = frame.height;
159
160 // 3a) Apply AEC AGC ROI if necessary
161 if(applyAECAGCrectLeft)
162 {
163 applyAECAGCrectLeft = false;
164 cap.setROIforAECAGC( sl_oc::video::CAM_SENS_POS::LEFT,
165 aecagc_roi_left.x, aecagc_roi_left.y,
166 aecagc_roi_left.width, aecagc_roi_left.height);
167 selectLeft=false;
168 selectRight=false;
169 }
170 if(applyAECAGCrectRight)
171 {
172 applyAECAGCrectRight = false;
173 cap.setROIforAECAGC( sl_oc::video::CAM_SENS_POS::RIGHT,
174 aecagc_roi_right.x, aecagc_roi_right.y,
175 aecagc_roi_right.width, aecagc_roi_right.height);
176 selectLeft=false;
177 selectRight=false;
178 }
179
180 // ----> If the frame is valid we can display it
181 if(frame.data!=nullptr && frame.timestamp!=last_ts)
182 {
183 not_a_new_frame=0;
184#if 0
185 // ----> Video Debug information
186
187 std::cout << std::setprecision(9) << "[" << frame.frame_id << "] Ts: " << static_cast<double>(frame.timestamp)/1e9 << " sec" << std::endl;
188 if( last_ts!=0 )
189 {
190 double dt = (frame.timestamp - last_ts)/1e9;
191 std::cout << std::setprecision(9) << " * dT: " << dt << " sec - FPS: " << 1./dt << std::endl;
192 }
193
194 // <---- Video Debug information
195#endif
196 last_ts = frame.timestamp;
197
198 // ----> Conversion from YUV 4:2:2 to BGR for visualization
199 cv::Mat frameYUV = cv::Mat( frame.height, frame.width, CV_8UC2, frame.data );
200 cv::Mat frameBGR;
201 cv::cvtColor(frameYUV,frameBGR,cv::COLOR_YUV2BGR_YUYV);
202 // <---- Conversion from YUV 4:2:2 to BGR for visualization
203
204 // 4.c) Show frame
205 showImage( win_name, frameBGR, params.res );
206 }
207 else if(frame.timestamp==last_ts)
208 {
209 not_a_new_frame++;
210 std::cout << "Not a new frame #" << not_a_new_frame << std::endl;
211
212 if( not_a_new_frame>=(3000/frame_timeout_msec)) // Lost connection for 5 seconds
213 {
214 std::cout << "Camera connection lost. Closing..." << std::endl;
215 break;
216 }
217 }
218 // <---- If the frame is valid we can display it
219
220 // ----> Keyboard handling
221 int key = cv::waitKey( 5 );
222
223 if( key != -1 )
224 {
225 if(key=='q' || key=='Q') // Quit
226 break;
227 else
228 handleKeyboard( cap, key );
229 }
230 // <---- Keyboard handling
231 }
232
233 return EXIT_SUCCESS;
234}

236// Handle Keyboard
237void handleKeyboard( sl_oc::video::VideoCapture &cap, int key )
238{
239 if(key >= '0' && key <= '9')
240 {
241 int value = key - '0';
242 setControlValue( cap, value );
243 return;
244 }
245
246 switch(key)
247 {
248#ifdef SENSOR_LOG_AVAILABLE
249 case 'L':
250 {
251 logging = !logging;
252 cap.enableAecAgcSensLogging( logging, 5 );
253 std::cout << std::string("*** AEC/AGC registers logging: ") << (logging?std::string("ENABLED"):std::string("DISABLED")) << std::endl;
254 }
255 break;
256
257 case 'f':
258 case 'F':
259 {
260 bool res = cap.resetAGCAECregisters();
261 std::cout << std::string("*** AEC/AGC registers reset: ") << (res?std::string("OK"):std::string("KO")) << std::endl;
262 }
263 break;
264#endif
265
266 case 'l':
267 {
268 bool led;
269 cap.toggleLED( &led );
270 std::cout << std::string(" * LED STATUS: ") << (led?std::string("ON"):std::string("OFF")) << std::endl;
271 }
272 break;
273
274 case 'b':
275 setActiveControl( cap, Brightness );
276 break;
277
278 case 'S':
279 setActiveControl( cap, Sharpness );
280 break;
281
282 case 'c':
283 setActiveControl( cap, Contrast );
284 break;
285
286 case 'h':
287 setActiveControl( cap, Hue );
288 break;
289
290 case 's':
291 setActiveControl( cap, Saturation );
292 break;
293
294 case 'w':
295 setActiveControl( cap, WhiteBalance );
296 break;
297
298 case 'g':
299 setActiveControl( cap, Gamma );
300 break;
301
302 case 'e':
303 setActiveControl( cap, Exposure );
304 break;
305
306 case 'G':
307 setActiveControl( cap, Gain );
308 break;
309
310 case 'a':
311 toggleAutomaticControl( cap );
312 break;
313
314 case 'r':
315 case 'R':
316 resetControls(cap);
317 updateAllCtrlValues(cap);
318
319 std::cout << "All control settings are reset to default values" << std::endl;
320 break;
321
322 case '+':
323 case 171:
324 changeControlValue(cap,true);
325 break;
326
327 case '-':
328 case 173:
329 changeControlValue(cap,false);
330 break;
331
332 case '?':
333 std::cout << "COMMANDS HELP" << std::endl;
334 std::cout << " * 'q' or 'Q' -> Quit the example" << std::endl;
335 std::cout << " * '?' -> This help menu" << std::endl;
336 std::cout << " * 'l' -> Toggle camera led" << std::endl;
337 std::cout << " * 'b' -> Brightness control" << std::endl;
338 std::cout << " * 's' -> Saturation control" << std::endl;
339 std::cout << " * 'c' -> Contrast control" << std::endl;
340 std::cout << " * 'h' -> Hue control" << std::endl;
341 std::cout << " * 'S' -> Sharpness control" << std::endl;
342 std::cout << " * 'w' -> White Balance control" << std::endl;
343 std::cout << " * 'g' -> Gamma control" << std::endl;
344 std::cout << " * 'e' -> Exposure control" << std::endl;
345 std::cout << " * 'G' -> Gain control" << std::endl;
346 std::cout << " * 'a' -> Toggle automatic for White Balance or Exposure and Gain" << std::endl;
347 std::cout << " * 'r' or 'R' -> Reset to default configuration" << std::endl;
348 std::cout << " * '+' -> Increase the current control value" << std::endl;
349 std::cout << " * '-' -> Decrease the current control value" << std::endl;
350 std::cout << " * '0' .. '9' -> Set the current control value" << std::endl;
351#ifdef SENSOR_LOG_AVAILABLE
352 std::cout << " * 'L' -> Toggle AGC/AEC registers logging" << std::endl;
353 std::cout << " * 'f' -> Fix AGC/AEC registers" << std::endl;
354#endif
355 }

359{

361 sharpness_val = cap.getSharpness();
362 contrast_val = cap.getContrast();

365 gamma_val = cap.getGamma();

370 whiteBalance_val = cap.getWhiteBalance();

385void handleMouse(int event, int x, int y, int, void*)
386{
387 switch (event)
388 {
389 case cv::EVENT_LBUTTONDOWN: // AEC AGC ROI drawing started
390 {
391 if(autoAECAGC && (activeControl==Gain || activeControl==Exposure))
392 {
393 origin_roi = cv::Point(x, y);
394 selectInProgress = true;
395 }
396 break;
397 }
398
399 case cv::EVENT_LBUTTONUP: // AECAGC ROI drawing completed
400 {
401 selectInProgress = false;
402
403 if(autoAECAGC && (activeControl==Gain || activeControl==Exposure))
404 {
405 if(selectLeft)
406 {
407 selectLeft=false;
408 applyAECAGCrectLeft = true;
409 }
410 if(selectRight)
411 {
412 selectRight=false;
413 applyAECAGCrectRight = true;
414 }
415 }
416 break;
417 }
418
419 case cv::EVENT_RBUTTONDOWN: // Reset AECAGC ROI with right button
420 {
421 //Reset selection
422 selectInProgress = false;
423 aecagc_roi_left = cv::Rect(0,0,img_w/2,img_h);
424 aecagc_roi_right = cv::Rect(0,0,img_w/2,img_h);
425 applyAECAGCrectLeft = true;
426 applyAECAGCrectRight = true;
427 break;
428 }
429 }
430
431 if(selectInProgress) // AECAGC ROI drawing in progress
432 {
433 x /= img_resize_factor;
434 y /= img_resize_factor;
435
436 int or_x = origin_roi.x/img_resize_factor;
437 int or_y = origin_roi.y/img_resize_factor;
438
439 y = MAX(y,0);
440 y = MIN(y,img_h);
441 if(or_x<img_w/2) // AECAGC ROI for the left image
442 {
443 x = MAX(x,0);
444 x = MIN(x,img_w/2-1);
445 selectLeft = true;
446
447 aecagc_roi_left.x = MIN(x, or_x);
448 aecagc_roi_left.y = MIN(y, or_y);
449 aecagc_roi_left.width = abs(x - or_x) + 1;
450 aecagc_roi_left.height = abs(y - or_y) + 1;
451 }
452 else // AECAGC ROI for the right image
453 {
454 or_x -= img_w/2;
455 x -= img_w/2;
456 x = MAX(x,0);
457 x = MIN(x,img_w/2-1);
458 selectRight = true;
459
460 aecagc_roi_right.x = MIN(x, or_x);
461 aecagc_roi_right.y = MIN(y, or_y);
462 aecagc_roi_right.width = abs(x - or_x) + 1;
463 aecagc_roi_right.height = abs(y - or_y) + 1;
464 }
465 }

581// '+' or '-' pressed
582void changeControlValue( sl_oc::video::VideoCapture &cap, bool increase )
583{
584 int curValue=0;
585 switch( activeControl )
586 {
587 case Brightness:
588 curValue = cap.getBrightness();
589 brightness_val = curValue;
590 break;
591
592 case Contrast:
593 curValue = cap.getContrast();
594 contrast_val = curValue;
595 break;
596
597 case Hue:
598 curValue = cap.getHue();
599 hue_val = curValue;
600 break;
601
602 case Saturation:
603 curValue = cap.getSaturation();
604 saturation_val = curValue;
605 break;
606
607 case Gain:
608 {
609 int curValueLeft = cap.getGain(sl_oc::video::CAM_SENS_POS::LEFT);
610 int curValueRight = cap.getGain(sl_oc::video::CAM_SENS_POS::RIGHT);
611
612 if(increase)
613 {
614 cap.setGain(sl_oc::video::CAM_SENS_POS::LEFT,++curValueLeft);
615 cap.setGain(sl_oc::video::CAM_SENS_POS::RIGHT,++curValueRight);
616 }
617 else
618 {
619 cap.setGain(sl_oc::video::CAM_SENS_POS::LEFT,--curValueLeft);
620 cap.setGain(sl_oc::video::CAM_SENS_POS::RIGHT,--curValueRight);;
621 }
622
623 gain_val_left = cap.getGain(sl_oc::video::CAM_SENS_POS::LEFT);
624 gain_val_right = cap.getGain(sl_oc::video::CAM_SENS_POS::RIGHT);
625
626 std::cout << "New Left Gain value: " << (int)gain_val_left << std::endl;
627 std::cout << "New Right Gain value: " << (int)gain_val_right << std::endl;
628
629 autoAECAGC = cap.getAECAGC();
630 }
631 break;
632
633 case Exposure:
634 {
635 int curValueLeft = cap.getExposure(sl_oc::video::CAM_SENS_POS::LEFT);
636 int curValueRight = cap.getExposure(sl_oc::video::CAM_SENS_POS::RIGHT);
637
638 if(increase)
639 {
640 cap.setExposure(sl_oc::video::CAM_SENS_POS::LEFT,++curValueLeft);
641 cap.setExposure(sl_oc::video::CAM_SENS_POS::RIGHT,++curValueRight);
642 }
643 else
644 {
645 cap.setExposure(sl_oc::video::CAM_SENS_POS::LEFT,--curValueLeft);
646 cap.setExposure(sl_oc::video::CAM_SENS_POS::RIGHT,--curValueRight);;
647 }
648
649 exposure_val_left = cap.getExposure(sl_oc::video::CAM_SENS_POS::LEFT);
650 exposure_val_right = cap.getExposure(sl_oc::video::CAM_SENS_POS::RIGHT);
651
652 std::cout << "New Left Exposure value: " << (int)exposure_val_left << std::endl;
653 std::cout << "New Right Exposure value: " << (int)exposure_val_right << std::endl;
654
655 autoAECAGC = cap.getAECAGC();
656 }
657 break;
658
659 case WhiteBalance:
660 cap.setAutoWhiteBalance(false);
661 curValue = cap.getWhiteBalance();
662 whiteBalance_val = curValue;
663 autoWB = cap.getAutoWhiteBalance();
664 break;
665
666 case Sharpness:
667 curValue = cap.getSharpness();
668 sharpness_val = curValue;
669 break;
670
671 case Gamma:
672 curValue = cap.getGamma();
673 gamma_val = curValue;
674 break;
675 }
676
677 if(activeControl==WhiteBalance)
678 {
679 if(increase)
680 curValue += 100;
681 else
682 curValue -= 100;
683
684 cap.setWhiteBalance( curValue );
685
686 whiteBalance_val = cap.getWhiteBalance();
687 std::cout << "New White Balance value: " << whiteBalance_val << std::endl;
688 }
689 else if(activeControl != Gain && activeControl != Exposure)
690 {
691 if(increase)
692 setControlValue( cap, ++curValue);
693 else
694 setControlValue( cap, --curValue);
695 }
696}

698// 'a' or 'A' pressed to enable automatic WhiteBalanse or Gain/Exposure
699void toggleAutomaticControl( sl_oc::video::VideoCapture &cap )
700{
701 if(activeControl == WhiteBalance)
702 {
703 bool curValue = cap.getAutoWhiteBalance();
704 cap.setAutoWhiteBalance( !curValue );
705 autoWB = cap.getAutoWhiteBalance();
706
707 std::cout << "Automatic White Balance control: " << ((!curValue)?"ENABLED":"DISABLED") << std::endl;
708 }
709
710 if(activeControl == Exposure || activeControl == Gain )
711 {
712 bool curValue = cap.getAECAGC();
713 cap.setAECAGC( !curValue );
714 autoAECAGC = cap.getAECAGC();
715
716 std::cout << "Automatic Exposure and Gain control: " << ((!curValue)?"ENABLED":"DISABLED") << std::endl;
717 }
718}

720// Rescale the images according to the selected resolution to better display them on screen
721void showImage( std::string name, cv::Mat& img, sl_oc::video::RESOLUTION res )
722{
723 cv::Mat resized;
724 switch(res)
725 {
726 default:
727 case sl_oc::video::RESOLUTION::VGA:
728 img_resize_factor = 1.0;
729 resized = img;
730 break;
731 case sl_oc::video::RESOLUTION::HD720:
732 name += " [Resize factor 0.6]";
733 img_resize_factor = 0.6;
734 cv::resize( img, resized, cv::Size(), img_resize_factor, img_resize_factor );
735 break;
736 case sl_oc::video::RESOLUTION::HD1080:
737 case sl_oc::video::RESOLUTION::HD2K:
738 name += " [Resize factor 0.4]";
739 img_resize_factor = 0.4;
740 cv::resize( img, resized, cv::Size(), img_resize_factor, img_resize_factor );
741 break;
742 }
743
744 if(autoAECAGC && (activeControl==Gain || activeControl==Exposure))
745 {
