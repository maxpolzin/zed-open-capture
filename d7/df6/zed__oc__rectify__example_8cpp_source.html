138// Rescale the images according to the selected resolution to better display them on screen
139void showImage( std::string name, cv::Mat& img, sl_oc::video::RESOLUTION res )
140{
141 cv::Mat resized;
142 switch(res)
143 {
144 default:
145 case sl_oc::video::RESOLUTION::VGA:
146 resized = img;
147 break;
148 case sl_oc::video::RESOLUTION::HD720:
149 name += " [Resize factor 0.6]";
150 cv::resize( img, resized, cv::Size(), 0.6, 0.6 );
151 break;
152 case sl_oc::video::RESOLUTION::HD1080:
153 case sl_oc::video::RESOLUTION::HD2K:
154 name += " [Resize factor 0.4]";
155 cv::resize( img, resized, cv::Size(), 0.4, 0.4 );
156 break;
157 }
158
159 cv::imshow( name, resized );
160}
